class DinamicFood:public Food
{
	string name;
	double price;
	Food*food;
public:
	DinamicFood(Food*food,string name,double prise)
	{
		this->food = food;
		this->name = name;
		this->price = prise;
	}
	~DinamicFood()
	{
		delete this->food;
	}
	string getName()
	{
		return food->getName() + " " + this->name;
	}
	double getPrice()
	{
		return food->getPrice() + this->price;
	}
};
